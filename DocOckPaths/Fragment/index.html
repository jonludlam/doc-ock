OCaml path fragments for specifying module substitutions

type kind = Kind.fragment
type sort = [
| `Root
| `Branch
]
type ('a, 'b, 'c) raw =
| Resolved : ('a, 'b, 'c) Resolved.raw ‑> ('a, 'b, 'c) raw
| Dot : 'a signature * string ‑> ('a, [< kind ], [< sort Branch ]) raw
type ('a, 'b) t = ('a, 'b, [ `Branch ]) raw
